A proposta implementar a soluo aproximada para a a funo de probabilidade acumulada normal padro $\Phi(y)$, de acordo com o apresentado no Anexo F do livro de referncia.
O cdigo abaixo apresenta a construo das aproximaes de $\Phi(y)$ para os intervalos $0 \leq y \leq \infty$ e $-\infty \leq y \leq 0$

import numpy as np
import matplotlib.pyplot as plt

# Valores do parmetro p_i
p_i = [0.231641900, 0.319381530, -0.356563782, 1.781477937, -1.821255978, 1.330274429]

# Funo w 
def w (y):
    return 1 / (1 + (p_i[0] * abs(y)))

# Funo z
def z (w):
    return (w * (p_i[1] + w * (p_i[2] + w * (p_i[3] + w * (p_i[4] + w * p_i[5])))))

# Aproximao analtica para a funo de probabilidade acumulada
# Para y negativo
def phi_neg (z , y):
    return (z / np.sqrt( 2 * np.pi)) * np.exp(- (y**2 / 2))

# Para y positivo
def phi_pos (z , y):
    return 1 - (z / np.sqrt( 2 * np.pi)) * np.exp(- (y**2 / 2))

# Para 0 < u <= 0.5
def y_1 (u):
    z = np.sqrt(np.log(1 / (u ** 2)))
    return -z - ((p[0] + z * (p[1] + z * (p[2] + z * (p[3] + z * (p[4]))))) / (q[0] + z * (q[1] + z * (q[2] + z * (q[3] + z * (q[4]))))))

# Para 0.5 <= u < 1
def y_2 (u):
    z = np.sqrt(np.log (1 / ((1 - u) ** 2)))
    return z + ((p[0] + z * (p[1] + z * (p[2] + z * (p[3] + z * (p[4]))))) / (q[0] + z * (q[1] + z * (q[2] + z * (q[3] + z * (q[4]))))))

No cdigo abaixo, temos uma estrutura de data members, na qual possvel calcular os momentos da distribuio (mdia, varincia, desvio-padro, coeficiente de variao, skewness e kurtosis) dado os parmetros.
Alm disso, h uma estrutura de member functions que calcula as funes $PDF$, $CDF$ e $CDF^{-1}$ e tambm calcula os parmetros dados os momentos.

from scipy import stats as st 
import numpy as np
import matplotlib.pyplot as plt

# Data members
class variavel_aleatoria:
    # Funo para identificar qual  a distribuio, o nome da varivel e o simbolo da distribuio
    def __init__(self, distribuicao: str, nome: str = "", simbolo: str =""):
        # Identificao
        self.nome = nome
        self.simbolo = simbolo
        self.distribuicao = distribuicao  

        # Lista de argumentos
        self.parametros = []
        self.objeto = None

        # Momentos da varivel
        self.media = np.nan 
        self.variancia = np.nan
        self.desvio = np.nan
        self.cv = np.nan # Coeficiente de variao
        self.skewness = np.nan
        self.kurtosis = np.nan

        # Distribuies implementadas
        self.distribuicoes = {
            'normal' : st.norm,
            'lognormal' : st.lognorm,
            'gumbel_max': st.gumbel_r,
            'gumbel_min': st.gumbel_l,
        }

        # Aqui as distribuies contempladas so atribuidas ao componente objeto
        self.objeto = self.distribuicoes[self.distribuicao]

        # Aqui os parmetros de cada distribuio so definidos e os momentos calculados a partir dos parmetros
    def conjunto_parametros (self, *params):
        self.parametros = params
        self.calculo_momentos()
    
        # Aqui os momentos so calculados a partir dos parametros
    def calculo_momentos(self):
        m, v, sk, k = self.objeto.stats(*self.parametros, moments = 'mvsk')

        # Armazenamento dos momentos nas variveis
        self.media = float(m)
        self.variancia = float(v)
        self.desvio = np.sqrt(self.variancia)
        self.skewness = float(sk)
        self.kurtosis = float(k)

        if self.media != 0:
            self.cv = self.desvio / self.media
        else:
            self.cv = np.nan
    
    # Aqui calcula-se os parametros de cada distribuio dado os momentos (mdia e desvio padro)
    def calculo_parametros (self, media_dada: float, desvio_dado: float):
        mu = media_dada
        sigma = desvio_dado

        if self.distribuicao =='normal':
            self.conjunto_parametros(mu, sigma)
        
        elif self.distribuicao == 'lognormal':
            zeta = np.sqrt(np.log(1.0 + (sigma / mu) ** 2))
            lam = np.log(mu) - (0.5 * (zeta ** 2))
            scale = np.exp(lam)
            self.conjunto_parametros(zeta, loc=0, scale=scale)

        elif self.distribuicao in ['gumbel_max', 'gumbel_min']:
            mu = media_dada
            sigma = desvio_dado
            gamma = 0.5772156649 #Constante de Euler-Mascheroni
            beta =  np.pi / (sigma * np.sqrt(6))

            if self.distribuicao == 'gumbel_max':
                mu_calc = mu - (gamma / beta)
            else:
                mu_calc = mu + (gamma / beta)
            self.conjunto_parametros(mu_calc, beta)
    
    # Agora vamos construir as funes fundamentais (PDF, CDF, Inversa)
    def PDF (self, x: float) -> float:
        if self.objeto:
            return self.objeto.pdf(x, *self.parametros)
        return np.nan
    
    def CDF (self, x: float) -> float:
        if self.objeto:
            return self.objeto.cdf(x, *self.parametros)
        return np.nan
    
    def InversaCDF (self, p: float) -> float:
        if self.objeto:
            return self.objeto.ppf(p, *self.parametros)
        return np.nan
